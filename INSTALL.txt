git clone http://github.com/rsletten/dot ~/dot
git clone https://github.com/rsletten/oh-my-zsh ~/.oh-my-zsh
cd ~/dot
git pull && git submodule init && git submodule update && git submodule status
ln -s ~/dot/robs.zsh-theme ~/.oh-my-zsh/themes
ln -s ~/dot/vim ~/.vim
ln -s ~/dot/vimrc ~/.vimrc
ln -s ~/dot/zshrc ~/.zshrc
ln -s ~/dot/tmux.conf ~/.tmux.conf
ln -s ~/dot/Xwinrc ~/.Xwinrc
ln -s ~/dot/startxwinrc ~/.startxwinrc
ln -s ~/dot/gitconfig ~/.gitconfig

# for ssh public key
git remote set-url origin git@github.com:rsletten/dot.git

# for https
git remote set-url origin https://github.com/rsletten/dot.git

# adding a vim module
cd dot
git submodule add https://github.com/scrooloose/syntastic.git vim/bundle/syntastic
git commit -a
git push

# removing a vim module
cd dot
git submodule deinit vim/bundle/vim-puppet
rmdir vim/bundle/vim-puppet
git commit -a
git push
